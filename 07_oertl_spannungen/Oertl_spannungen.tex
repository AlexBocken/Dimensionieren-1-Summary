        \subsubsection{Kerben \& Oberflächen ($\sigma_{WK}\quad2\rightarrow3$)}
            \[K_{\textrm{WK}}=\left(\left[1+\frac{1}{\widetilde{K}_F}\left(\frac{1}{K_R}-1\right)\right]\frac{1}{K_v}\right)\frac{1}{\eta_{\sigma}}\]
